// --- Utils ---
const formatCurrency = (amount: number) => {
  return new Intl.NumberFormat('ko-KR', { style: 'currency', currency: 'KRW' }).format(amount);
};

const formatCompactCurrency = (amount: number) => {
  if (amount >= 100000000) {
    return `${(amount / 100000000).toFixed(1)}억`;
  }
  if (amount >= 10000) {
    return `${(amount / 10000).toFixed(0)}만`;
  }
  return amount.toLocaleString();
};

const formatDate = (timestamp: any) => {
  if (!timestamp) return '방금 전';
  try {
    const date = timestamp.toDate ? timestamp.toDate() : new Date(timestamp);
    return new Intl.DateTimeFormat('ko-KR', {
      year: '2-digit',
      month: 'numeric',
      day: 'numeric',
      hour: '2-digit',
      minute: '2-digit',
      hour12: false
    }).format(date);
  } catch (e) {
    return '-';
  }
};

const compressImage = (file: File): Promise<string> => {
  return new Promise((resolve, reject) => {
    const reader = new FileReader();
    reader.readAsDataURL(file);
    reader.onload = (event) => {
      const img = new Image();
      img.src = event.target?.result as string;
      img.onload = () => {
        const canvas = document.createElement('canvas');
        const MAX_WIDTH = 500; 
        const scaleSize = MAX_WIDTH / img.width;
        canvas.width = MAX_WIDTH;
        canvas.height = img.height * scaleSize;
        
        const ctx = canvas.getContext('2d');
        if (ctx) {
          ctx.drawImage(img, 0, 0, canvas.width, canvas.height);
          resolve(canvas.toDataURL('image/jpeg', 0.8));
        } else {
          reject(new Error("Canvas context failed"));
        }
      };
      img.onerror = (error) => reject(error);
    };
    reader.onerror = (error) => reject(error);
  });
};

const exportToCSV = (items: InventoryItem[]) => {
  try {
    const BOM = '\uFEFF';
    const headers = ['카테고리,품목명,현재고,단위,적정재고,개당원가,합계금액,최종수정일,상태'];
    
    const rows = items.map(item => {
      const status = item.quantity < item.minThreshold ? '재고부족' : '정상';
      const categoryMap: Record<string, string> = { 'finished': '완제품', 'raw': '원재료', 'sub': '부자재' };
      const categoryName = categoryMap[item.category] || item.category;
      const cost = item.costPrice || 0;
      const totalVal = (item.quantity * cost);
      const dateStr = formatDate(item.updatedAt);
      
      return `"${categoryName}","${item.name}",${item.quantity},"${item.unit}",${item.minThreshold},${cost},${totalVal},"${dateStr}","${status}"`;
    });

    const csvContent = BOM + headers.join(",") + "\n" + rows.join("\n");
    const blob = new Blob([csvContent], { type: 'text/csv;charset=utf-8;' });
    const url = URL.createObjectURL(blob);
    const link = document.createElement("a");
    link.setAttribute("href", url);
    
    const date = new Date();
    const safeDate = `${date.getFullYear()}-${String(date.getMonth() + 1).padStart(2, '0')}-${String(date.getDate()).padStart(2, '0')}`;
    link.setAttribute("download", `허브FnC_자산현황_${safeDate}.csv`);
    
    document.body.appendChild(link);
    link.click();
    document.body.removeChild(link);
    URL.revokeObjectURL(url);
  } catch (error) {
    console.error("Export Error:", error);
  }
};

export default function App() {
  const [user, setUser] = useState<any>(null);
  const [isLocked, setIsLocked] = useState(true);
  const [items, setItems] = useState<InventoryItem[]>([]);
  const [loading, setLoading] = useState(true);
  const [searchTerm, setSearchTerm] = useState('');
  const [categoryFilter, setCategoryFilter] = useState<'all' | Category>('all');
  const [isAddModalOpen, setIsAddModalOpen] = useState(false);
  const [editingId, setEditingId] = useState<string | null>(null);
  const [editForm, setEditForm] = useState<Partial<InventoryItem>>({});
  
  const [deleteTargetId, setDeleteTargetId] = useState<string | null>(null);
  // 로그아웃 모달 상태 추가
  const [isLogoutModalOpen, setIsLogoutModalOpen] = useState(false);

  // Auth
  useEffect(() => {
    const initAuth = async () => {
      try {
        if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
          await signInWithCustomToken(auth, __initial_auth_token);
        } else {
          await signInAnonymously(auth);
        }
      } catch (error) {
        console.error("Auth Error:", error);
      }
    };
    initAuth();
    const unsubscribe = onAuthStateChanged(auth, setUser);
    return () => unsubscribe();
  }, []);

  // Data Fetching
  useEffect(() => {
    if (!user) return;
    
    const q = query(collection(db, 'artifacts', appId, 'public', 'data', 'inventory'));
    
    const unsubscribe = onSnapshot(q, (snapshot) => {
      const loadedItems: InventoryItem[] = [];
      snapshot.forEach((doc) => {
        const data = doc.data();
        loadedItems.push({ 
          id: doc.id, 
          ...data,
          costPrice: data.costPrice || 0 
        } as InventoryItem);
      });
      
      loadedItems.sort((a, b) => {
          if (a.category !== b.category) return a.category.localeCompare(b.category);
          return a.name.localeCompare(b.name);
      });
      
      setItems(loadedItems);
      setLoading(false);
    }, (error) => {
      console.error("Firestore Error:", error);
      setLoading(false);
    });

    return () => unsubscribe();
  }, [user]);

  // Calculations
  const totalAssetValue = useMemo(() => items.reduce((sum, item) => sum + (item.quantity * item.costPrice), 0), [items]);
  const finishedAssetValue = useMemo(() => items.filter(i => i.category === 'finished').reduce((sum, i) => sum + (i.quantity * i.costPrice), 0), [items]);
  const rawAssetValue = useMemo(() => items.filter(i => i.category === 'raw').reduce((sum, i) => sum + (i.quantity * i.costPrice), 0), [items]);
  const subAssetValue = useMemo(() => items.filter(i => i.category === 'sub').reduce((sum, i) => sum + (i.quantity * i.costPrice), 0), [items]);

  // Operations
  const handleAddItem = async (newItem: Omit<InventoryItem, 'id' | 'updatedAt'>) => {
    if (!user) throw new Error("No user");
    
    await addDoc(collection(db, 'artifacts', appId, 'public', 'data', 'inventory'), {
      ...newItem,
      updatedAt: serverTimestamp()
    });
  };

  const handleExport = () => {
    if (items.length === 0) {
      return;
    }
    exportToCSV(items);
  };

  const handleDeleteRequest = (id: string) => {
    setDeleteTargetId(id);
  };

  const confirmDelete = async () => {
    if (!deleteTargetId || !user) return;
    try {
      await deleteDoc(doc(db, 'artifacts', appId, 'public', 'data', 'inventory', deleteTargetId));
      setDeleteTargetId(null);
    } catch (error) {
      console.error("Delete Error:", error);
    }
  };

  // 로그아웃 요청 (모달 열기)
  const handleLogoutRequest = () => {
    setIsLogoutModalOpen(true);
  };

  // 로그아웃 확정
  const confirmLogout = () => {
    setIsLogoutModalOpen(false);
    setIsLocked(true);
  };

  const startEditing = (item: InventoryItem) => {
    setEditingId(item.id);
    setEditForm({ ...item });
  };

  const saveEditing = async () => {
    if (!editingId) return;
    try {
      await updateDoc(doc(db, 'artifacts', appId, 'public', 'data', 'inventory', editingId), {
        ...editForm,
        updatedAt: serverTimestamp()
      });
      setEditingId(null);
      setEditForm({});
    } catch (error) {
      console.error("Update Error:", error);
    }
  };

  const updateQuantity = async (id: string, current: number, delta: number) => {
    const newQty = Math.max(0, current + delta);
    try {
      await updateDoc(doc(db, 'artifacts', appId, 'public', 'data', 'inventory', id), {
        quantity: newQty,
        updatedAt: serverTimestamp()
      });
    } catch (error) {
      console.error("Qty Update Error:", error);
    }
  };

  const filteredItems = useMemo(() => {
    return items.filter(item => {
      const matchesSearch = item.name.toLowerCase().includes(searchTerm.toLowerCase());
      const matchesCategory = categoryFilter === 'all' || item.category === categoryFilter;
      return matchesSearch && matchesCategory;
    });
  }, [items, searchTerm, categoryFilter]);

  const lowStockItems = items.filter(item => item.quantity < item.minThreshold);
  
  const getCategoryLabel = (cat: Category) => {
    switch(cat) {
      case 'raw': return { text: '원재료', color: 'bg-emerald-100 text-emerald-800' };
      case 'sub': return { text: '부자재', color: 'bg-amber-100 text-amber-800' };
      case 'finished': return { text: '완제품', color: 'bg-blue-100 text-blue-800' };
      default: return { text: '기타', color: 'bg-gray-100' };
    }
  };

  if (isLocked) {
    return <LoginScreen onLogin={() => setIsLocked(false)} />;
  }

  return (
    <div className="min-h-screen bg-gray-50 text-gray-900 font-sans flex flex-col pb-24">
      {/* Top Navigation */}
      <header className="bg-white shadow-sm border-b sticky top-0 z-40">
        <div className="max-w-7xl mx-auto px-4 h-16 flex justify-between items-center">
          <div className="flex items-center gap-2">
            <div className="bg-blue-900 p-2 rounded-lg text-white shadow-md">
              <Factory size={20} />
            </div>
            <div className="flex flex-col">
              <span className="text-lg font-bold leading-none text-gray-900">허브F&C</span>
              <span className="text-[10px] text-gray-500 font-medium tracking-wider">SMART INVENTORY</span>
            </div>
          </div>
          <div className="flex items-center gap-2">
            <button 
              onClick={handleExport}
              className="bg-white hover:bg-gray-50 text-gray-700 border border-gray-200 px-3 py-2 rounded-lg font-bold shadow-sm flex items-center gap-2 transition active:scale-95"
              title="엑셀로 다운로드"
            >
              <Download size={18} />
              <span className="hidden sm:inline text-sm">목록 저장</span>
            </button>
            <button 
              onClick={() => setIsAddModalOpen(true)}
              className="bg-blue-900 hover:bg-blue-800 active:bg-blue-950 text-white px-4 py-2 rounded-lg font-bold shadow-md flex items-center gap-2 transition transform active:scale-95"
            >
              <Plus size={18} />
              <span className="hidden sm:inline">품목 등록</span>
              <span className="sm:hidden">등록</span>
            </button>
          </div>
        </div>
      </header>

      <main className="max-w-7xl mx-auto px-4 py-6 w-full flex-1">
        
        {loading && (
          <div className="flex flex-col items-center justify-center py-20">
            <div className="animate-spin rounded-full h-12 w-12 border-4 border-blue-200 border-t-blue-900 mb-4"></div>
            <p className="text-gray-500 font-medium animate-pulse">데이터 불러오는 중...</p>
          </div>
        )}

        {!loading && (
          <>
            {/* Asset Summary Cards */}
            <div className="grid grid-cols-2 lg:grid-cols-4 gap-3 mb-6">
              <div className="bg-white p-4 rounded-xl shadow-sm border border-gray-100">
                <p className="text-xs text-gray-500 font-medium mb-1">총 자산 (전체)</p>
                <p className="text-xl font-bold text-gray-900">{formatCompactCurrency(totalAssetValue)}<span className="text-xs text-gray-400 font-normal ml-1">원</span></p>
              </div>
               <div className="bg-blue-50 p-4 rounded-xl shadow-sm border border-blue-100">
                <p className="text-xs text-blue-600 font-medium mb-1">완제품 자산</p>
                <p className="text-xl font-bold text-blue-900">{formatCompactCurrency(finishedAssetValue)}</p>
              </div>
               <div className="bg-emerald-50 p-4 rounded-xl shadow-sm border border-emerald-100">
                <p className="text-xs text-emerald-600 font-medium mb-1">원재료 자산</p>
                <p className="text-xl font-bold text-emerald-900">{formatCompactCurrency(rawAssetValue)}</p>
              </div>
              <div className="bg-amber-50 p-4 rounded-xl shadow-sm border border-amber-100">
                <p className="text-xs text-amber-600 font-medium mb-1">부자재 자산</p>
                <p className="text-xl font-bold text-amber-900">{formatCompactCurrency(subAssetValue)}</p>
              </div>
            </div>

            {/* Alert Banner */}
            {lowStockItems.length > 0 && (
              <div className="bg-red-50 border border-red-200 p-4 mb-6 rounded-xl flex items-start gap-3 animate-bounce-slight shadow-sm">
                <AlertTriangle className="text-red-600 shrink-0 mt-0.5" size={20} />
                <div>
                  <h4 className="font-bold text-red-800 text-sm">재고 확인 필요</h4>
                  <p className="text-red-700 text-xs mt-0.5">
                    현재 <span className="font-bold">{lowStockItems.length}</span>개의 품목 재고가 기준치 미만입니다.
                  </p>
                </div>
              </div>
            )}

            {/* Filters */}
            <div className="sticky top-16 z-30 bg-gray-50 pb-4 pt-2 -mx-2 px-2">
              <div className="flex gap-2 overflow-x-auto pb-3 px-1 hide-scrollbar">
                {[
                  { id: 'all', label: '전체', value: totalAssetValue, color: 'gray' },
                  { id: 'finished', label: '완제품', value: finishedAssetValue, color: 'blue' },
                  { id: 'raw', label: '원재료', value: rawAssetValue, color: 'emerald' },
                  { id: 'sub', label: '부자재', value: subAssetValue, color: 'amber' }
                ].map((cat) => (
                  <button 
                    key={cat.id}
                    onClick={() => setCategoryFilter(cat.id as any)}
                    className={`flex flex-col items-center px-4 py-2 rounded-xl text-sm font-bold whitespace-nowrap transition shadow-sm border ${
                      categoryFilter === cat.id 
                        ? `bg-${cat.color === 'gray' ? 'gray-800' : cat.color + '-600'} text-white border-transparent scale-105` 
                        : `bg-white text-gray-600 border-gray-200 hover:bg-gray-50`
                    }`}
                  >
                    <span>{cat.label}</span>
                    <span className={`text-[10px] font-normal opacity-80 mt-0.5`}>
                      {formatCompactCurrency(cat.value)}
                    </span>
                  </button>
                ))}
              </div>
              
              <div className="relative mt-2">
                <Search className="absolute left-4 top-1/2 transform -translate-y-1/2 text-gray-400" size={18} />
                <input 
                  type="text" 
                  placeholder="품목 이름으로 검색..." 
                  className="w-full pl-11 pr-4 py-3 bg-white border border-gray-200 rounded-xl focus:ring-2 focus:ring-blue-500 outline-none text-base shadow-sm"
                  value={searchTerm}
                  onChange={(e) => setSearchTerm(e.target.value)}
                />
                {searchTerm && (
                  <button 
                    onClick={() => setSearchTerm('')}
                    className="absolute right-3 top-1/2 transform -translate-y-1/2 text-gray-400 p-1"
                  >
                    <X size={16} />
                  </button>
                )}
              </div>
            </div>

            {/* Grid */}
            <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-5">
              {filteredItems.map((item) => {
                const isLowStock = item.quantity < item.minThreshold;
                const catStyle = getCategoryLabel(item.category);
                const isEditing = editingId === item.id;
                const totalItemValue = item.quantity * item.costPrice;

                return (
                  <div key={item.id} className={`bg-white rounded-2xl border shadow-sm overflow-hidden flex flex-col transition-all duration-200 ${isLowStock ? 'ring-2 ring-red-100 border-red-200' : 'border-gray-100 hover:shadow-md'}`}>
                    {/* Image Area */}
                    <div className="relative h-44 bg-gray-50 flex items-center justify-center overflow-hidden group">
                      {item.imageUrl ? (
                        <img src={item.imageUrl} alt={item.name} className="w-full h-full object-contain p-2 transition-transform duration-500 group-hover:scale-105" />
                      ) : (
                        <div className="flex flex-col items-center text-gray-300">
                          <ImageIcon size={40} />
                          <span className="text-xs mt-2 font-medium">사진 없음</span>
                        </div>
                      )}
                      
                      <span className={`absolute top-3 left-3 px-2.5 py-1 rounded-lg text-[10px] font-bold uppercase tracking-wide shadow-sm backdrop-blur-sm ${catStyle.color}`}>
                        {catStyle.text}
                      </span>
                      
                      {!isEditing && (
                        <button 
                          onClick={() => startEditing(item)}
                          className="absolute top-3 right-3 bg-white/90 p-2 rounded-full shadow-md text-gray-600 hover:text-blue-600 transition active:scale-90"
                        >
                          <Edit2 size={16} />
                        </button>
                      )}
                    </div>

                    {/* Content Area */}
                    <div className="p-5 flex-1 flex flex-col">
                      {isEditing ? (
                        <div className="space-y-3 mb-2 animate-fadeIn">
                          <div>
                            <label className="text-xs text-gray-400 block mb-1">이름</label>
                            <input 
                              type="text" 
                              value={editForm.name} 
                              onChange={(e) => setEditForm({...editForm, name: e.target.value})}
                              className="w-full border border-gray-300 rounded p-2 text-sm font-bold focus:ring-2 focus:ring-blue-500 outline-none"
                            />
                          </div>
                          <div className="flex gap-2">
                             <div className="w-1/2">
                              <label className="text-xs text-gray-400 block mb-1">단가(원)</label>
                              <input 
                                type="number" 
                                value={editForm.costPrice} 
                                onChange={(e) => setEditForm({...editForm, costPrice: Number(e.target.value)})}
                                className="w-full border border-gray-300 rounded p-2 text-sm focus:ring-2 focus:ring-blue-500 outline-none"
                              />
                            </div>
                            <div className="w-1/2">
                              <label className="text-xs text-gray-400 block mb-1">단위</label>
                              <input 
                                type="text" 
                                value={editForm.unit} 
                                onChange={(e) => setEditForm({...editForm, unit: e.target.value})}
                                className="w-full border border-gray-300 rounded p-2 text-sm focus:ring-2 focus:ring-blue-500 outline-none"
                              />
                            </div>
                          </div>
                        </div>
                      ) : (
                        <div className="mb-4">
                          <h3 className="font-bold text-gray-900 text-xl leading-tight mb-1">{item.name}</h3>
                          
                          <div className="flex items-center gap-1 text-xs text-gray-400 mb-2">
                            <Clock size={10} />
                            <span>{formatDate(item.updatedAt)}</span>
                          </div>
                          
                          <div className="flex flex-col gap-1 mb-2 bg-gray-50 p-2 rounded-lg border border-gray-100">
                             <div className="flex justify-between items-center text-xs text-gray-500">
                                <span>개당 단가</span>
                                <span>{item.costPrice?.toLocaleString()}원</span>
                             </div>
                             <div className="flex justify-between items-center text-sm font-bold text-gray-800 border-t border-gray-200 pt-1 mt-1">
                                <span>합계</span>
                                <span className="text-blue-900">{formatCurrency(totalItemValue)}</span>
                             </div>
                          </div>

                          <div className="flex items-center gap-2 text-xs text-gray-500">
                             <span className="bg-gray-100 px-2 py-0.5 rounded text-gray-600">기준: {item.minThreshold}{item.unit}</span>
                             {isLowStock && <span className="text-red-500 font-bold flex items-center gap-1"><AlertTriangle size={12} /> 부족함</span>}
                          </div>
                        </div>
                      )}

                      <div className="mt-auto pt-4 border-t border-gray-50 flex items-center justify-between">
                        {isEditing ? (
                          <div className="flex w-full gap-2">
                             <button onClick={saveEditing} className="flex-1 bg-green-600 text-white py-2 rounded-lg text-sm font-bold shadow-sm active:bg-green-700">완료</button>
                             <button onClick={() => setEditingId(null)} className="flex-1 bg-gray-100 text-gray-700 py-2 rounded-lg text-sm font-medium active:bg-gray-200">취소</button>
                          </div>
                        ) : (
                          <>
                            <div className="flex items-center gap-1 bg-gray-50 rounded-lg p-1 border border-gray-100">
                              <button 
                                onClick={() => updateQuantity(item.id, item.quantity, -1)}
                                className="w-9 h-9 rounded-md bg-white border border-gray-200 shadow-sm flex items-center justify-center text-gray-600 active:bg-gray-50 active:scale-95 transition"
                              >
                                -
                              </button>
                              <div className="w-16 text-center">
                                <span className={`text-lg font-bold font-mono ${isLowStock ? 'text-red-600' : 'text-gray-900'}`}>
                                  {item.quantity}
                                </span>
                                <span className="text-xs text-gray-400 block -mt-1">{item.unit}</span>
                              </div>
                              <button 
                                onClick={() => updateQuantity(item.id, item.quantity, 1)}
                                className="w-9 h-9 rounded-md bg-white border border-blue-200 shadow-sm flex items-center justify-center text-blue-600 active:bg-blue-50 active:scale-95 transition"
                              >
                                +
                              </button>
                            </div>
                            <button 
                              onClick={() => handleDeleteRequest(item.id)}
                              className="text-gray-300 hover:text-red-500 hover:bg-red-50 p-2 rounded-lg transition"
                            >
                              <Trash2 size={18} />
                            </button>
                          </>
                        )}
                      </div>
                    </div>
                  </div>
                );
              })}
              
              {/* Empty State */}
              {filteredItems.length === 0 && !loading && (
                <div className="col-span-full py-12 text-center flex flex-col items-center text-gray-400">
                  <div className="bg-gray-50 p-4 rounded-full mb-3">
                    <Search size={32} />
                  </div>
                  <p>조건에 맞는 품목이 없습니다.</p>
                </div>
              )}
            </div>
          </>
        )}
      </main>

      {/* Footer */}
      <footer className="py-8 text-center bg-gray-50 mt-auto">
          <button 
            onClick={handleLogoutRequest}
            className="inline-flex items-center gap-2 text-gray-400 hover:text-red-600 bg-white border border-gray-200 px-6 py-2 rounded-full text-sm font-medium shadow-sm transition hover:shadow-md hover:border-red-100"
          >
            <LogOut size={14} /> 로그아웃
          </button>
      </footer>

      <AddItemModal 
        isOpen={isAddModalOpen} 
        onClose={() => setIsAddModalOpen(false)} 
        onAdd={handleAddItem}
      />
      
      <DeleteConfirmModal 
        isOpen={!!deleteTargetId}
        onClose={() => setDeleteTargetId(null)}
        onConfirm={confirmDelete}
      />

      <LogoutConfirmModal 
        isOpen={isLogoutModalOpen}
        onClose={() => setIsLogoutModalOpen(false)}
        onConfirm={confirmLogout}
      />
    </div>
  );
}
